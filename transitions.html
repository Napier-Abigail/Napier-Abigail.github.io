CSS Transitions

transition: <property> <duration> <timing> <delay>

property = which css value are you wanting to place the transition on

duration = how long you want it to take for the transition to complete

timing = type of timing for transition (ease, linear, ease-in, ease-out, ease-in-out)

delay = wait this long before starting transition

ease = fast - slow - fast

linear = constant

ease-in = slow-fast

ease-out = fast-slow

ease-in-out = slow-fast-slow
